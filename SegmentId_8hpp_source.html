69 inline SegmentId SegmentId::id_of_parent() const {
70  ASSERT(0 != refinement_level_, "Called on root refinement level!");
71  // right-shifting index_ by 1 gives us floor(index_/2).
72  // The parent has half as many segments as the child.
73  return SegmentId(refinement_level_ - 1, index_ >> 1);
74 }
75 
76 inline SegmentId SegmentId::id_of_child(Side side) const {
77  // We cannot ASSERT on the maximum level because it's only known at runtime
78  // and only known elsewhere in the code, not by SegmentId. I.e. SegmentId is
79  // too low-level to know about this.
80  // The child has twice as many segments as the parent, so for a particular
81  // parent segment, there is both an upper and lower child segment.
82  if (Side::Lower == side) {
83  // left-shifting index_ by 1 gives us index*2.
84  return SegmentId(refinement_level_ + 1, index_ << 1);
85  }
86  // left-shifting index_ by 1, followed by OR with 1, gives us index*2 +1.
87  return SegmentId(refinement_level_ + 1, 1 | (index_ << 1));
88 }

95 inline double SegmentId::endpoint(Side side) const {
96  if (Side::Lower == side) {
97  return -1.0 + (2.0 * index_) / two_to_the(refinement_level_);
98  }
99  return -1.0 + (2.0 * index_ + 2.0) / two_to_the(refinement_level_);
100 }
