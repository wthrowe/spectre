28 // applies the function f(out, it) to each item from begin to end, separated
29 // by commas and surrounded by parens
30 template <typename ForwardIt, typename Func>
31 inline void print_helper(std::ostream& out, ForwardIt&& begin, ForwardIt&& end,
32  Func f) {
33  out << "(";
34  if (begin != end) {
35  while (true) {
36  f(out, begin++);
37  if (begin == end) {
38  break;
39  }
40  out << ",";
41  }
42  }
43  out << ")";
44 }
