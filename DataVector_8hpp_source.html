367 template <typename VT, bool VF>
368 DataVector& DataVector::operator=(const blaze::Vector<VT, VF>& expression) {
369  if (owning_) {
370  size_ = (~expression).size();
371  owned_data_ = InternalStorage_t(size_);
372  data_ = decltype(data_){owned_data_.data(), size_};
373  data_ = expression;
374  } else {
375  ASSERT((~expression).size() == size(), "Must copy into same size");
376  data_ = expression;
377  }
378  return *this;
379 }
